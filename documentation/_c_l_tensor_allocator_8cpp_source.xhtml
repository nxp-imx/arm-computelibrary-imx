37 std::unique_ptr<ICLMemoryRegion> allocate_region(cl::Context context, size_t size, cl_uint alignment)
38 {
39  // Try fine-grain SVM
40  std::unique_ptr<ICLMemoryRegion> region = support::cpp14::make_unique<CLFineSVMMemoryRegion>(context,
41  CL_MEM_READ_WRITE | CL_MEM_SVM_FINE_GRAIN_BUFFER,
42  size,
43  alignment);
44 
45  // Try coarse-grain SVM in case of failure
46  if(region != nullptr && region->ptr() == nullptr)
47  {
48  region = support::cpp14::make_unique<CLCoarseSVMMemoryRegion>(context, CL_MEM_READ_WRITE, size, alignment);
49  }
50  // Try legacy buffer memory in case of failure
51  if(region != nullptr && region->ptr() == nullptr)
52  {
53  region = support::cpp14::make_unique<CLBufferMemoryRegion>(context, CL_MEM_ALLOC_HOST_PTR | CL_MEM_READ_WRITE, size);
54  }
55  return region;
56 }

74 void CLTensorAllocator::allocate()
75 {
76  if(_associated_memory_group == nullptr)
77  {
78  if(_memory.region() != nullptr && _memory.cl_region()->cl_data().get() != nullptr)
79  {
80  // Memory is already allocated. Reuse it if big enough, otherwise fire an assertion
81  ARM_COMPUTE_ERROR_ON_MSG(info().total_size() > _memory.region()->size(),
82  "Reallocation of a bigger memory region is not allowed!");
83  }
84  else
85  {
86  // Perform memory allocation
87  _memory.set_owned_region(allocate_region(CLScheduler::get().context(), info().total_size(), 0));
88  }
89  }
90  else
91  {
92  _associated_memory_group->finalize_memory(_owner, _memory, info().total_size());
93  }
94  info().set_is_resizable(false);
95 }
