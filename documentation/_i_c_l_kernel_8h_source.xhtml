328 template <typename T, unsigned int dimension_size>
329 void ICLKernel::add_array_argument(unsigned &idx, const ICLArray<T> *array, const Strides &strides, unsigned int num_dimensions, const Window &window)
330 {
331  ARM_COMPUTE_ERROR_ON(array == nullptr);
332 
333  // Calculate offset to the start of the window
334  unsigned int offset_first_element = 0;
335 
336  for(unsigned int n = 0; n < num_dimensions; ++n)
337  {
338  offset_first_element += window[n].start() * strides[n];
339  }
340 
341  unsigned int idx_start = idx;
342  _kernel.setArg(idx++, array->cl_buffer());
343 
344  for(unsigned int dimension = 0; dimension < dimension_size; dimension++)
345  {
346  _kernel.setArg<cl_uint>(idx++, strides[dimension]);
347  _kernel.setArg<cl_uint>(idx++, strides[dimension] * window[dimension].step());
348  }
349 
350  _kernel.setArg<cl_uint>(idx++, offset_first_element);
351 
352  ARM_COMPUTE_ERROR_ON_MSG(idx_start + num_arguments_per_array<dimension_size>() != idx,
353  "add_%dD_array_argument() is supposed to add exactly %d arguments to the kernel", dimension_size, num_arguments_per_array<dimension_size>());
354  ARM_COMPUTE_UNUSED(idx_start);
355 }
