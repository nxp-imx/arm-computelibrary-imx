50 Status NEDeconvolutionLayer::validate(const ITensorInfo *input, const ITensorInfo *weights, const ITensorInfo *bias, const ITensorInfo *output, const PadStrideInfo &info,
51  unsigned int inner_border_right, unsigned int inner_border_top)
52 {
53  ARM_COMPUTE_RETURN_ERROR_ON_NULLPTR(input, weights, output);
54  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(input, 1, DataType::F32);
55  ARM_COMPUTE_RETURN_ERROR_ON_DATA_TYPE_CHANNEL_NOT_IN(weights, 1, DataType::F32);
56  ARM_COMPUTE_RETURN_ERROR_ON(weights->dimension(0) != weights->dimension(1));
57  ARM_COMPUTE_RETURN_ERROR_ON(weights->dimension(0) < 1);
58  ARM_COMPUTE_RETURN_ERROR_ON(!info.padding_is_symmetric());
59 
60  const unsigned int stride_x = info.stride().first;
61  const unsigned int stride_y = info.stride().second;
62 
63  ARM_COMPUTE_RETURN_ERROR_ON_MSG(inner_border_right > stride_x - 1, "inner_border_right must be smaller than stride_x");
64  ARM_COMPUTE_RETURN_ERROR_ON_MSG(inner_border_top > stride_y - 1, "inner_border_top must be smaller than stride_y");
65 
66  auto out_dims = deconvolution_output_dimensions(input->dimension(0), input->dimension(1), weights->dimension(0), weights->dimension(1),
67  info.pad().first, info.pad().second, stride_x, stride_y);
68 
69  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, weights);
70 
71  if(bias != nullptr)
72  {
73  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, bias);
74  }
75 
76  if(output->tensor_shape().total_size() > 0)
77  {
78  ARM_COMPUTE_RETURN_ERROR_ON_MISMATCHING_DATA_TYPES(input, output);
79 
80  const TensorShape output_shape = compute_deconvolution_output_shape(out_dims, *input, *weights);
81 
82  ARM_COMPUTE_RETURN_ERROR_ON_MSG(output->dimension(Window::DimX) != output_shape.x(), "Output's width is invalid.");
83  ARM_COMPUTE_RETURN_ERROR_ON_MSG(output->dimension(Window::DimY) != output_shape.y(), "Output's height is invalid.");
84  ARM_COMPUTE_RETURN_ERROR_ON_MSG(output->dimension(Window::DimZ) != output_shape.z(), "Output's depth is invalid.");
85  }
86 
87  unsigned int padx = 0;
88  unsigned int pady = 0;
89  const TensorShape scale_out_shape = compute_deconvolution_upsampled_shape(*input, *weights, stride_x, stride_y, inner_border_right, inner_border_top, out_dims, padx, pady);
90  TensorInfo scale_out_info(input->clone()->set_is_resizable(true).reset_padding().set_tensor_shape(scale_out_shape));
91  const PadStrideInfo conv_info(1, 1, 0, 0, 0, 0, DimensionRoundingType::CEIL);
92 
93  for(size_t i = 2; i < Coordinates::num_max_dimensions; ++i)
94  {
95  ARM_COMPUTE_RETURN_ERROR_ON(input->dimension(i) != scale_out_info.dimension(i));
96  }
97 
98  ARM_COMPUTE_RETURN_ON_ERROR(NEConvolutionLayer::validate(&scale_out_info, weights, bias, output, conv_info, WeightsInfo()));
99 
100  return Status{};
101 }

103 void NEDeconvolutionLayer::configure(ITensor *input, const ITensor *weights, const ITensor *bias, ITensor *output, const PadStrideInfo &info,
104  unsigned int inner_border_right, unsigned int inner_border_top)
105 {
106  ARM_COMPUTE_ERROR_ON_NULLPTR(input, weights, output);
107 
108  _input = input;
109  _original_weights = weights;
110  _info = info;
111  _inner_border = std::make_pair(inner_border_right, inner_border_top);
112  _is_prepared = false;
113 
114  const unsigned int stride_x = info.stride().first;
115  const unsigned int stride_y = info.stride().second;
116 
117  _weights_flipped.allocator()->init(TensorInfo(weights->info()->tensor_shape(), 1, weights->info()->data_type()));
118  _flip_weights.configure(weights, &_weights_flipped);
119 
120  auto out_dims = deconvolution_output_dimensions(input->info()->dimension(0), input->info()->dimension(1), weights->info()->dimension(0), weights->info()->dimension(1),
121  info.pad().first, info.pad().second, stride_x, stride_y);
122 
123  const TensorShape output_shape = compute_deconvolution_output_shape(out_dims, *input->info(), *weights->info());
124  // Output auto initialization if not yet initialized
125  auto_init_if_empty(*output->info(), output_shape, 1, input->info()->data_type(), input->info()->quantization_info());
126 
127  // Perform validation step
128  ARM_COMPUTE_ERROR_THROW_ON(NEDeconvolutionLayer::validate(input->info(), weights->info(), bias == nullptr ? nullptr : bias->info(), output->info(), info, inner_border_right, inner_border_top));
129 
130  _memory_group.manage(&_scaled_output);
131 
132  // Find the upsampled dimensions and the padding needed for the convolution with stride 1 in order to match output shape
133  unsigned int padx = 0;
134  unsigned int pady = 0;
135  const TensorShape scale_out_shape = compute_deconvolution_upsampled_shape(*input->info(), *weights->info(), stride_x, stride_y, inner_border_right, inner_border_top, out_dims, padx, pady);
136 
137  TensorInfo scale_out_info(scale_out_shape, 1, input->info()->data_type(), input->info()->quantization_info());
138  _scaled_output.allocator()->init(scale_out_info);
139 
140  const PadStrideInfo upsample_info(stride_x, stride_y, padx / 2, pady / 2);
141  _upsample_f.configure(input, &_scaled_output, upsample_info, inner_border_right, inner_border_top);
142 
143  // setup the function to convolve the upscaled output
144  const PadStrideInfo conv_info(1, 1, 0, 0, 0, 0, DimensionRoundingType::CEIL);
145  _conv_f.configure(&_scaled_output, &_weights_flipped, bias, output, conv_info);
146  _scaled_output.allocator()->allocate();
147 }

170 void NEDeconvolutionLayer::prepare()
171 {
172  if(!_is_prepared)
173  {
174  ARM_COMPUTE_ERROR_ON(!_original_weights->is_used());
175 
176  // Run weights flipping and mark original weights tensor as unused
177  _weights_flipped.allocator()->allocate();
178  NEScheduler::get().schedule(&_flip_weights, Window::DimZ);
179  _original_weights->mark_as_unused();
180 
181  // Prepare convolution
182  _conv_f.prepare();
183 
184  if(!_weights_flipped.is_used())
185  {
186  _weights_flipped.allocator()->free();
187  }
188 
189  _is_prepared = true;
190  }
