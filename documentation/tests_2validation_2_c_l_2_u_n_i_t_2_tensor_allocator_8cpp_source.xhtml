68 TEST_CASE(ImportMemoryBuffer, framework::DatasetMode::ALL)
69 {
70  // Init tensor info
71  const TensorInfo info(TensorShape(24U, 16U, 3U), 1, DataType::F32);
72 
73  // Allocate memory buffer
74  const size_t total_size = info.total_size();
75  auto buf = cl::Buffer(CLScheduler::get().context(), CL_MEM_READ_WRITE, total_size);
76 
77  // Negative case : Import nullptr
78  CLTensor t1;
79  t1.allocator()->init(info);
80  ARM_COMPUTE_EXPECT(!bool(t1.allocator()->import_memory(cl::Buffer())), framework::LogLevel::ERRORS);
81  ARM_COMPUTE_EXPECT(t1.info()->is_resizable(), framework::LogLevel::ERRORS);
82 
83  // Negative case : Import memory to a tensor that is memory managed
84  CLTensor t2;
85  CLMemoryGroup mg;
86  t2.allocator()->set_associated_memory_group(&mg);
87  ARM_COMPUTE_EXPECT(!bool(t2.allocator()->import_memory(buf)), framework::LogLevel::ERRORS);
88  ARM_COMPUTE_EXPECT(t2.info()->is_resizable(), framework::LogLevel::ERRORS);
89 
90  // Negative case : Invalid buffer size
91  CLTensor t3;
92  const TensorInfo info_neg(TensorShape(32U, 16U, 3U), 1, DataType::F32);
93  t3.allocator()->init(info_neg);
94  ARM_COMPUTE_EXPECT(!bool(t3.allocator()->import_memory(buf)), framework::LogLevel::ERRORS);
95  ARM_COMPUTE_EXPECT(t3.info()->is_resizable(), framework::LogLevel::ERRORS);
96 
97  // Positive case : Set raw pointer
98  CLTensor t4;
99  t4.allocator()->init(info);
100  ARM_COMPUTE_EXPECT(bool(t4.allocator()->import_memory(buf)), framework::LogLevel::ERRORS);
101  ARM_COMPUTE_EXPECT(!t4.info()->is_resizable(), framework::LogLevel::ERRORS);
102  ARM_COMPUTE_EXPECT(t4.cl_buffer().get() == buf.get(), framework::LogLevel::ERRORS);
103  t4.allocator()->free();
104  ARM_COMPUTE_EXPECT(t4.info()->is_resizable(), framework::LogLevel::ERRORS);
105  ARM_COMPUTE_EXPECT(t4.cl_buffer().get() != buf.get(), framework::LogLevel::ERRORS);
106 }

108 TEST_CASE(ImportMemoryMalloc, framework::DatasetMode::ALL)
109 {
110  // Check if import extension is supported
111  if(!device_supports_extension(CLKernelLibrary::get().get_device(), "cl_arm_import_memory_host"))
112  {
113  return;
114  }
115  else
116  {
117  const ActivationLayerInfo act_info(ActivationLayerInfo::ActivationFunction::RELU);
118  const TensorShape shape = TensorShape(24U, 16U, 3U);
119  const DataType data_type = DataType::F32;
120 
121  // Create tensor
122  const TensorInfo info(shape, 1, data_type);
123  CLTensor tensor;
124  tensor.allocator()->init(info);
125 
126  // Create and configure activation function
127  CLActivationLayer act_func;
128  act_func.configure(&tensor, nullptr, act_info);
129 
130  // Allocate and import tensor
131  const size_t total_size_in_elems = tensor.info()->tensor_shape().total_size();
132  const size_t total_size_in_bytes = tensor.info()->total_size();
133  const size_t alignment = CLKernelLibrary::get().get_device().getInfo<CL_DEVICE_GLOBAL_MEM_CACHELINE_SIZE>();
134  size_t space = total_size_in_bytes + alignment;
135  auto raw_data = support::cpp14::make_unique<uint8_t[]>(space);
136 
137  void *aligned_ptr = raw_data.get();
138  support::cpp11::align(alignment, total_size_in_bytes, aligned_ptr, space);
139 
140  cl::Buffer wrapped_buffer(import_malloc_memory_helper(aligned_ptr, total_size_in_bytes));
141  ARM_COMPUTE_EXPECT(bool(tensor.allocator()->import_memory(wrapped_buffer)), framework::LogLevel::ERRORS);
142  ARM_COMPUTE_EXPECT(!tensor.info()->is_resizable(), framework::LogLevel::ERRORS);
143 
144  // Fill tensor
145  std::uniform_real_distribution<float> distribution(-5.f, 5.f);
146  std::mt19937 gen(library->seed());
147  auto *typed_ptr = reinterpret_cast<float *>(aligned_ptr);
148  for(unsigned int i = 0; i < total_size_in_elems; ++i)
149  {
150  typed_ptr[i] = distribution(gen);
151  }
152 
153  // Execute function and sync
154  act_func.run();
155  CLScheduler::get().sync();
156 
157  // Validate result by checking that the input has no negative values
158  for(unsigned int i = 0; i < total_size_in_elems; ++i)
159  {
160  ARM_COMPUTE_EXPECT(typed_ptr[i] >= 0, framework::LogLevel::ERRORS);
161  }
162 
163  // Release resources
164  tensor.allocator()->free();
165  ARM_COMPUTE_EXPECT(tensor.info()->is_resizable(), framework::LogLevel::ERRORS);
166  }
167 }
